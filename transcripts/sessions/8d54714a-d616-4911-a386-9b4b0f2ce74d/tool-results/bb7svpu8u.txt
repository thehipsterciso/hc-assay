===CI===
name: ci

on:
  push:
    branches: [main]
  pull_request:

# Least privilege (#H-008): the default GITHUB_TOKEN can be repo-write; this workflow only reads
# the code and uploads artifacts, so pin the token to read-only at the top level. Any job needing
# more must opt in explicitly with its own `permissions:`.
permissions:
  contents: read

jobs:
  # Dependency-free core: the engine must lint, type-check, and test with NO extras installed
  # (optional backends are import-guarded; integration tests skip). This guards ADR-0006.
  core:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        python-version: ["3.11", "3.12", "3.13", "3.14"]
    steps:
      - uses: actions/checkout@34e114876b0b11c390a56381ad16ebd13914f8d5 # v4
      - uses: actions/setup-python@a26af69be951a213d495a4c3e4e4022e16d87065 # v5
        with:
          python-version: ${{ matrix.python-version }}
      - name: Install (no extras, hash-pinned)
        run: |
          # Install the dev toolchain from the hashed, audited core lockfile so tests run against
          # EXACTLY the versions pip-audit scanned and the SBOM describes — not a live PyPI
          # resolution where a malicious patch release could execute under test (#F-004). The
          # core lock carries only the `dev` extra (no backends), preserving the ADR-0006 guard.
          python -m pip install --upgrade pip
          pip install --require-hashes -r requirements-core.lock
          pip install --require-hashes -r requirements-build.lock  # hashed build backend (#J-003)
          pip install --no-build-isolation --no-deps -e .
      - name: Lint (ruff)
        run: ruff check src tests
      - name: Format check (ruff)
        run: ruff format --check src tests
      - name: Type-check (mypy)
        run: mypy
      - name: Test (pytest)
        run: pytest -q

  # Extras lane: install the real backends and run the integration suite so the backend code
  # paths actually execute, and mypy type-checks against the REAL (not ignored) library types.
  # Service-dependent tests (ollama/qdrant/postgres) skip when the service is absent in CI; the
  # extra-only tests (langgraph graph, mlflow lifecycle, OTel, numpy primitives, methodology
  # e2e) run for real.
  integration:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_PASSWORD: postgres
          POSTGRES_DB: assay
        ports: ["5432:5432"]
        options: >-
          --health-cmd "pg_isready -U postgres"
          --health-interval 5s --health-timeout 5s --health-retries 10
      qdrant:
        image: qdrant/qdrant:v1.12.4
        ports: ["6333:6333"]
    env:
      # loopback host keeps the data-sovereignty guard satisfied; creds are redacted on error
      ASSAY_TEST_POSTGRES_URL: postgresql://[REDACTED]@localhost:5432/assay
    steps:
      - uses: actions/checkout@34e114876b0b11c390a56381ad16ebd13914f8d5 # v4
      - uses: actions/setup-python@a26af69be951a213d495a4c3e4e4022e16d87065 # v5
        with:
          python-version: "3.12"
      - name: Install (all extras, hash-pinned)
        run: |
          # Install from the hashed all-extras lockfile so the integration suite executes against
          # exactly the audited, pinned versions — closing the window where a compromised patch
          # release within a pyproject range could run during the real-backend tests (#F-004).
          python -m pip install --upgrade pip
          pip install --require-hashes -r requirements.lock
          pip install --require-hashes -r requirements-build.lock  # hashed build backend (#J-003)
          pip install --no-build-isolation --no-deps -e .
      - name: Type-check against real backend types (mypy)
        run: mypy
      - name: Verify service containers are reachable (no silent skip)
        # #H-009: the live Postgres/Qdrant tests SKIP when the service is unreachable, so a
        # misconfigured/never-started container would let the integration job report GREEN without
        # actually exercising the backends. Fail loudly here if either service isn't up, so the
        # gate cannot pass vacuously.
        run: |
          set -euo pipefail
          for i in $(seq 1 30); do
            pg_isready -h localhost -p 5432 -U postgres && break || sleep 2
            [ "$i" = "30" ] && { echo "::error::Postgres service never became ready"; exit 1; }
          done
          for i in $(seq 1 30); do
            curl -fsS http://localhost:6333/readyz && break || sleep 2
            [ "$i" = "30" ] && { echo "::error::Qdrant service never became ready"; exit 1; }
          done
          echo "postgres + qdrant reachable"
      # Real backends now run: langgraph graph + interrupt/resume, mlflow lifecycle, OTel,
      # numpy primitives, the T3 drain/classification (synthetic SDK, no metered call), AND —
      # via the service containers — the live Postgres checkpointer + Qdrant vector store.
      # Ollama and the metered frontier tier are not in CI and skip honestly.
      - name: Integration + unit tests with coverage
        # --cov-fail-under ENFORCES a floor (#K-OPS-3): without it `pytest --cov` prints coverage
        # but exits 0 at any level, so the coverage "gate" was vacuous — a refactor deleting tests
        # or adding a large untested module would still pass green. The floor (90) sits below the
        # current ~97% with headroom for environment-dependent skips; ratchet it up over time.
        run: pytest -q --cov=assay_engine --cov-report=term-missing --cov-fail-under=90

  # Supply-chain: fail on known-vulnerable dependencies (a security/privacy engine must not ship
  # with advisories). The scan runs against the committed, fully-pinned, hashed lockfile
  # (requirements.lock) so the audit is reproducible across runs and matches what installs
  # (#108, #130, #131, #132, #145, #146). An allowlist file (.pip-audit-ignore, one advisory id
  # per line, '#' comments and blank lines allowed) covers accepted/unfixable ones.
  audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@34e114876b0b11c390a56381ad16ebd13914f8d5 # v4
      - uses: actions/setup-python@a26af69be951a213d495a4c3e4e4022e16d87065 # v5
        with:
          python-version: "3.12"
      - name: Install pip-audit
        # Pin the audit tooling to exact versions (#G-022): an unpinned pip-audit can change
        # advisory results / exit behavior between releases, making the supply-chain gate
        # non-reproducible (a green commit could fail tomorrow on a tooling bump) — the analogue
        # of the uv pin (#F-031).
        run: python -m pip install --upgrade pip && python -m pip install pip-audit==2.10.1
      - name: Lockfiles are in sync with pyproject (reproducibility guard)
        run: |
          # Pin uv to the exact version the lockfiles were generated with (#F-031): an unpinned
          # `uv` upgrade can change output format/ordering/annotations and produce spurious diffs
          # that fail this gate on every PR even when the lockfiles are current.
          python -m pip install uv==0.11.21
          # Pin the RESOLUTION to a fixed point-in-time index (#K-OPS-1). Without --exclude-newer,
          # uv resolves the newest in-range version of every (largely unconstrained, transitive)
          # dep at CI wall-clock time, so this "reproducibility" gate goes red the instant ANY
          # upstream publishes an in-range release — on PRs that touch no dependency. The date is
          # the single source of truth in .uv-exclude-newer (regenerate_lockfiles.sh reads the same
          # file), so the regen here matches the committed lock exactly. Bump that file to update.
          EXCLUDE_NEWER="$(tr -d '[:space:]' < .uv-exclude-newer)"
          uv pip compile pyproject.toml --all-extras --universal --generate-hashes --exclude-newer "$EXCLUDE_NEWER" -o /tmp/requirements.regen.lock
          # ignore the auto-generated header (timestamp/command line) when diffing
          diff <(grep -v '^#' requirements.lock) <(grep -v '^#' /tmp/requirements.regen.lock) \
            || { echo "::error::requirements.lock is stale — run: scripts/regenerate_lockfiles.sh"; exit 1; }
          # the core lockfile (dev extra only) backs the hash-pinned core lane (#F-004) and must
          # stay in sync too.
          uv pip compile pyproject.toml --extra dev --universal --generate-hashes --exclude-newer "$EXCLUDE_NEWER" -o /tmp/requirements-core.regen.lock
          diff <(grep -v '^#' requirements-core.lock) <(grep -v '^#' /tmp/requirements-core.regen.lock) \
            || { echo "::error::requirements-core.lock is stale — run: scripts/regenerate_lockfiles.sh"; exit 1; }
          # the build-backend lockfile (hatchling etc.) backs the hash-pinned --no-build-isolation
          # editable install (#J-003) and must stay in sync.
          printf 'hatchling\neditables\n' > /tmp/build-reqs.in
          uv pip compile /tmp/build-reqs.in --universal --generate-hashes --exclude-newer "$EXCLUDE_NEWER" -o /tmp/requirements-build.regen.lock
          diff <(grep -v '^#' requirements-build.lock) <(grep -v '^#' /tmp/requirements-build.regen.lock) \
            || { echo "::error::requirements-build.lock is stale — run: scripts/regenerate_lockfiles.sh"; exit 1; }
      - name: pip-audit (reproducible known-vulnerability scan)
        run: |
          set -euo pipefail
          # --strict: any dependency that cannot be collected/audited FAILS the gate (no silent
          # skips). --require-hashes: every pin must carry a hash. Build --ignore-vuln args as an
          # array so allowlist entries can't word-split or glob-expand, and read the file
          # newline-agnostically so the last line is honored even without a trailing newline.
          IGNORES=()
          if [ -f .pip-audit-ignore ]; then
            while IFS= read -r id || [ -n "$id" ]; do
              case "$id" in ''|\#*) continue ;; esac
              IGNORES+=(--ignore-vuln "$id")
            done < .pip-audit-ignore
          fi
          pip-audit --strict --require-hashes --progress-spinner off -r requirements.lock "${IGNORES[@]}"
      - name: SBOM (CycloneDX) from the lockfile
        run: |
          set -euo pipefail
          # Emit a machine-readable software bill of materials for the pinned dependency set so
          # downstream license/transitive monitoring has a stable artifact to track (#147). No
          # `|| true` mask (#F-012): a pip-audit failure must fail the step, and the SBOM must be
          # non-empty, so a broken/absent SBOM cannot pass CI green.
          # Apply the SAME allowlist as the gate (#G-015): pip-audit exits non-zero on ANY finding
          # regardless of output format, so without the allowlist an ACCEPTED advisory would fail
          # SBOM generation here even though the gate step passed — coupling SBOM emission to vuln
          # status and making the allowlist ineffective. The gate step already ran first, so a
          # NON-allowlisted advisory has failed the job before we reach here.
          IGNORES=()
          if [ -f .pip-audit-ignore ]; then
            while IFS= read -r id || [ -n "$id" ]; do
              case "$id" in ''|\#*) continue ;; esac
              IGNORES+=(--ignore-vuln "$id")
            done < .pip-audit-ignore
          fi
          pip-audit --progress-spinner off -r requirements.lock --format cyclonedx-json -o sbom.cyclonedx.json "${IGNORES[@]}"
          test -s sbom.cyclonedx.json
          echo "SBOM emitted ($(wc -c < sbom.cyclonedx.json) bytes)"
      - name: Upload SBOM artifact
        if: always()
        uses: actions/upload-artifact@ea165f8d65b6e75b540449e92b4886f43607fa02 # v4
        with:
          name: sbom-cyclonedx
          path: sbom.cyclonedx.json
          if-no-files-found: error
      - name: License gate (deny strong-copyleft in the pinned dependency set)
        run: |
          set -euo pipefail
          # Fail if any dependency carries a strong-copyleft license incompatible with the
          # Apache-2.0 distribution (#147). Scan the SAME hash-pinned set pip-audit audited, not a
          # live `pip install -e .[all]` resolution that could pull different versions than the
          # lockfile pins (#F-013). Advisory monitoring of the rest is handled by Dependabot.
          pip install --require-hashes -r requirements.lock
          pip install --require-hashes -r requirements-build.lock  # hashed build backend (#J-003)
          pip install --no-build-isolation --no-deps -e .
          python -m pip install pip-licenses==5.5.5  # pinned for reproducible license output (#G-022)
          pip-licenses --format=json > licenses.json  # scan the dep set, not system tooling (#J-006)
          python scripts/license_gate.py licenses.json
      - name: Upload license report
        if: always()
        uses: actions/upload-artifact@ea165f8d65b6e75b540449e92b4886f43607fa02 # v4
        with:
          name: license-report
          path: licenses.json
          if-no-files-found: warn

  # Single required status check (#F-014): branch protection should require ONLY `all-checks`.
  # It fans in every other job, so adding a new job automatically gates merges without anyone
  # remembering to update the branch-protection rule — and `audit`/`integration` can never be
  # silently dropped from the merge gate by requiring just `core`.
  all-checks:
    if: always()
    needs: [core, integration, audit]
    runs-on: ubuntu-latest
    steps:
      - name: Verify all required jobs succeeded
        run: |
          results="${{ needs.core.result }} ${{ needs.integration.result }} ${{ needs.audit.result }}"
          echo "job results: $results"
          for r in $results; do
            if [ "$r" != "success" ]; then
              echo "::error::a required job did not succeed ($r)"
              exit 1
            fi
          done
          echo "all required jobs succeeded"

"""Supply-chain / CI audit-gate regression guards (pass-2 #130, #131, #132, #145, #146, #147).

These guard configuration that lives in .github/workflows/ci.yml and the committed lockfile.
The CI YAML cannot be unit-imported, so the guards are twofold:
  1. config-level: assert the hardened invocation is present in ci.yml (a revert to the buggy
     form fails the test);
  2. behavioral: run the exact allowlist-parsing bash loop against fixtures to prove it is
     newline-agnostic and comment/blank tolerant (the original `while read -r id` form drops a
     final line that lacks a trailing newline).
"""

from __future__ import annotations

from pathlib import Path

import pytest

_ROOT = Path(__file__).resolve().parent.parent
_CI = _ROOT / ".github" / "workflows" / "ci.yml"
_LOCK = _ROOT / "requirements.lock"

# The hardened allowlist loop, lifted verbatim from ci.yml's audit job. Kept here so the
# behavioral test exercises the SAME logic the workflow runs.
_PARSE_LOOP = r"""
set -euo pipefail
IGNORES=()
if [ -f "$1" ]; then
  while IFS= read -r id || [ -n "$id" ]; do
    case "$id" in ''|\#*) continue ;; esac
    IGNORES+=(--ignore-vuln "$id")
  done < "$1"
fi
printf '%s\n' "${IGNORES[@]:-<none>}"
"""

# The ORIGINAL buggy loop, for the revert-discrimination assertion in the behavioral test.
_PARSE_LOOP_BUGGY = r"""
IGNORES=""
if [ -f "$1" ]; then
  while read -r id; do [ -n "$id" ] && IGNORES="$IGNORES --ignore-vuln $id"; done < "$1"
fi
printf '%s\n' "$IGNORES"
"""

def test_lockfile_is_fully_pinned_and_hashed():
    # #132/#146: every dependency line is an exact == pin carrying at least one sha256 hash, so the
    # audited set is reproducible and no over-wide range can drift the resolved version.
    body = [
        ln
        for ln in _LOCK.read_text(encoding="utf-8").splitlines()
        if ln and not ln.lstrip().startswith("#") and not ln.lstrip().startswith("--hash")
    ]
    pinned = [ln for ln in body if "==" in ln]
    assert len(pinned) >= 50, "lockfile looks empty/under-resolved"
    text = _LOCK.read_text(encoding="utf-8")
    assert "--hash=sha256:" in text, "lockfile must carry hashes (#132 --require-hashes)"
    # no loose comparators that admit a range in the LOCK (pyproject ranges stay permissive)
    for ln in pinned:
        assert ">=" not in ln and "<" not in ln.split(";")[0], f"lock pin not exact: {ln!r}"

def test_ci_allowlist_loop_is_robust_in_the_workflow_file():
    # #131/#145: guard ci.yml ITSELF (not only the behavioral copy) — the workflow must use the
    # newline-agnostic read and the quoted-array expansion, and must NOT contain the buggy forms.
    text = _ci_text()
    assert "read -r id || [ -n" in text, "ci.yml allowlist read is not newline-agnostic (#131)"
    assert "IGNORES+=(" in text and '"${IGNORES[@]}"' in text, (
        "ci.yml IGNORES not a quoted array (#145)"
    )
    # the original buggy string-accumulation form must be gone
    assert 'IGNORES="$IGNORES --ignore-vuln' not in text, (
        "buggy unquoted IGNORES string reintroduced"
    )
    assert "while read -r id; do [ -n" not in text, "buggy non-newline-agnostic loop reintroduced"
    # #F-052: the behavioral allowlist tests run a hardcoded copy of the loop against a tmp file,
    # so they cannot catch a change to the allowlist FILENAME in ci.yml. Pin the filename here.
    assert ".pip-audit-ignore" in text, "ci.yml allowlist filename changed (#F-052)"


def test_no_dependency_range_spans_multiple_majors():
    # #146: an over-wide range (e.g. arize-phoenix>=7.0,<18) admits ~10 majors and defeats the
    # reproducibility intent. Guard that each extra's pins keep floor and ceiling within one major.
    import re
    import tomllib

    pp = tomllib.loads((_ROOT / "pyproject.toml").read_text(encoding="utf-8"))
    extras = pp["project"]["optional-dependencies"]
    offenders = []
    for group, deps in extras.items():
        for spec in deps:
            m = re.search(r">=\s*(\d+)[\d.]*\s*,\s*<\s*(\d+)", spec)
            if m:
                floor_major, ceil_major = int(m.group(1)), int(m.group(2))
                if ceil_major - floor_major > 2:  # allow <floor+2; flag the wide ones
                    offenders.append(f"{group}: {spec} spans {ceil_major - floor_major} majors")
    assert not offenders, f"over-wide dependency ranges: {offenders}"

def _load_license_gate():
    import importlib.util

    spec = importlib.util.spec_from_file_location(
        "license_gate", _ROOT / "scripts" / "license_gate.py"
    )
    assert spec and spec.loader
    mod = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(mod)
    return mod


@pytest.mark.parametrize(
    "license_name,denied",
    [
        ("Server Side Public License", True),  # #F-015: SSPL prose form
        ("European Union Public Licence 1.2", True),  # #F-015: EUPL prose form
        ("GNU Affero General Public License v3", True),  # AGPL prose form
        ("SSPL", True),
        ("GPL-3.0", True),
        ("GNU Lesser General Public License v3 (LGPLv3)", False),  # weak copyleft permitted
        ("MIT", False),
        ("Apache-2.0", False),
        ("BSD-3-Clause", False),
    ],
)
def test_license_gate_behavioral_denies_strong_copyleft(tmp_path, license_name, denied):
    # #F-015 + #F-039: a BEHAVIORAL test (call main() on a synthetic report), not a string-presence
    # check — the prose-name SSPL/EUPL bypass and any logic regression are caught here.
    import json

    gate = _load_license_gate()
    report = [{"Name": "pkg", "Version": "1.0", "License": license_name}]
    p = tmp_path / "licenses.json"
    p.write_text(json.dumps(report), encoding="utf-8")
    rc = gate.main(str(p))
    assert rc == (1 if denied else 0), f"{license_name!r}: expected denied={denied}"

def test_license_gate_surfaces_unknown_licenses(tmp_path):
    # #H-007: an UNKNOWN/empty-license package must not pass SILENTLY (it could hide a copyleft) —

    gate = _load_license_gate()
    report = [
        {"Name": "mystery", "Version": "1", "License": "UNKNOWN"},

@pytest.mark.skipif(shutil.which("bash") is None, reason="bash required")
def test_allowlist_parse_is_newline_agnostic_and_comment_tolerant(tmp_path):
    # #131/#145: the last advisory id must be honored even with no trailing newline, comments and
    # blank lines skipped, and ids must not word-split/glob-expand.
    f = tmp_path / "ignore"
    f.write_bytes(b"# a comment\n\nGHSA-aaaa\nPYSEC-2024-last-no-newline")  # no trailing \n
    out = subprocess.run(
        ["bash", "-c", _PARSE_LOOP, "_", str(f)],
        capture_output=True,
        text=True,
        check=True,
    ).stdout
    assert "GHSA-aaaa" in out
    assert "PYSEC-2024-last-no-newline" in out, "final newline-less id dropped (#131)"
    assert "# a comment" not in out, "comment line leaked into ignores (#131)"

    # revert-discrimination: the original loop DROPS the final newline-less id.
    buggy = subprocess.run(
        ["bash", "-c", _PARSE_LOOP_BUGGY, "_", str(f)],
        capture_output=True,
        text=True,
        check=True,
    ).stdout
    assert "PYSEC-2024-last-no-newline" not in buggy, (
        "buggy loop unexpectedly kept the id — test no longer discriminates"
    )


@pytest.mark.skipif(shutil.which("bash") is None, reason="bash required")
def test_allowlist_ids_do_not_glob_expand(tmp_path):
    # #145: an allowlist line is passed as a single argv element even if it contains shell glob
    # metacharacters (it must never expand against the working directory).
    f = tmp_path / "ignore"
    f.write_text("GHSA-*\n", encoding="utf-8")
    (tmp_path / "GHSA-real-file").write_text("x", encoding="utf-8")
    out = subprocess.run(
        ["bash", "-c", f"cd {tmp_path} && " + _PARSE_LOOP, "_", str(f)],
        capture_output=True,
        text=True,
        check=True,
    ).stdout
    assert "GHSA-*" in out, "glob metacharacter expanded against cwd (#145)"
    assert "GHSA-real-file" not in out
